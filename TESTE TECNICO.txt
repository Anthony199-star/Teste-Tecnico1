BANCO DE DADOS:
CREATE DATABASE TesteTecnico;

USE TesteTecnico;

CREATE TABLE Associados (
    id_associado INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(200) NOT NULL,
    Cpf VARCHAR(11) NOT NULL UNIQUE,
    Data_nascimento DATETIME
);

CREATE TABLE Empresas (
    id_empresa INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(200) NOT NULL,
    CNPJ VARCHAR(14) NOT NULL UNIQUE
);

CREATE TABLE AssociadosEmpresas (
    empresaid INT NOT NULL,
    associadoid INT NOT NULL,
    FOREIGN KEY (empresaid) REFERENCES Empresas(id_empresa),
    FOREIGN KEY (associadoid) REFERENCES Associados(id_associado),
    PRIMARY KEY (empresaid, associadoid)
);


APP.JS:
// Importar módulos necessários
const express = require('express');
const mysql = require('mysql2');
const fileupload = require('express-fileupload');
const { engine } = require('express-handlebars');

// Configurar aplicação
const app = express();
app.use(fileupload());
app.use('/bootstrap', express.static('./node_modules/bootstrap/dist'));
app.use('/css', express.static('./css'));
app.engine('handlebars', engine());
app.set('view engine', 'handlebars');
app.set('views', './views');
app.use(express.json());
app.use(express.urlencoded({ extended: false }));

// Configurar conexão com o banco de dados
const conexao = mysql.createConnection({
    host: 'localhost',
    user: 'root',
    password: '',
    database: 'TesteTecnico'
});

// Testar conexão
conexao.connect(function(erro) {
    if (erro) throw erro;
    console.log('Conexão com o banco de dados estabelecida.');
});

// Rota principal
app.get('/', function(req, res) {
    res.redirect('/associados');
});

// Consulta com filtros - Associados
app.get('/associados', function(req, res) {
    const { Nome, Cpf, Data_nascimento } = req.query;
    let sql = 'SELECT * FROM associados WHERE 1=1';
    let params = [];
    if (Nome) sql += ' AND Nome LIKE ?', params.push('%' + Nome + '%');
    if (Cpf) sql += ' AND Cpf = ?', params.push(Cpf);
    if (Data_nascimento) sql += ' AND Data_nascimento = ?', params.push(Data_nascimento);
    conexao.query(sql, params, function(erro, retorno) {
        if (erro) res.status(500).send('Erro ao carregar a lista de associados.');
        else res.render('associados', { associados: retorno });
    });
});

// Consulta por ID - Associado
app.get('/associados/:id_associado', function(req, res) {
    const id_associado = req.params.id_associado;
    let sql = 'SELECT * FROM associados WHERE id_associado = ?';
    conexao.query(sql, [id_associado], function(erro, retorno) {
        if (erro) res.status(500).send('Erro ao carregar os dados do associado.');
        else res.send(retorno[0]);
    });
});

// Cadastro de Associado
app.post('/associados', function(req, res) {
    let { Nome, Cpf, Data_nascimento } = req.body;
    let sql = 'INSERT INTO associados (Nome, Cpf, Data_nascimento) VALUES (?, ?, ?)';
    conexao.query(sql, [Nome, Cpf, Data_nascimento], function(erro) {
        if (erro) res.status(500).send('Erro ao cadastrar o associado.');
        else res.redirect('/associados');
    });
});

// Alteração de Associado com vínculos
app.post('/editarassociados/:id_associado', function(req, res) {
    const id_associado = req.params.id_associado;
    const { Nome, Cpf, Data_nascimento, EmpresasVinculadas } = req.body;
    let sql = 'UPDATE associados SET Nome = ?, Cpf = ?, Data_nascimento = ? WHERE id_associado = ?';
    conexao.query(sql, [Nome, Cpf, Data_nascimento, id_associado], function(erro) {
        if (erro) res.status(500).send('Erro ao atualizar o associado.');
        else {
            let sqlDelete = 'DELETE FROM associadosempresas WHERE associadoid = ?';
            conexao.query(sqlDelete, [id_associado], function() {
                let sqlInsert = 'INSERT INTO associadosempresas (empresaid, associadoid) VALUES (?, ?)';
                EmpresasVinculadas.forEach(empresaid => {
                    conexao.query(sqlInsert, [empresaid, id_associado]);
                });
            });
            res.redirect('/associados');
        }
    });
});

// Exclusão de Associado
app.get('/removerassociado/:id_associado', function(req, res) {
    const id_associado = req.params.id_associado;
    let sqlDeleteRel = 'DELETE FROM associadosempresas WHERE associadoid = ?';
    conexao.query(sqlDeleteRel, [id_associado], function() {});
    let sqlDelete = 'DELETE FROM associados WHERE id_associado = ?';
    conexao.query(sqlDelete, [id_associado], function(erro) {
        if (erro) res.status(500).send('Erro ao remover o associado.');
        else res.redirect('/associados');
    });
});
